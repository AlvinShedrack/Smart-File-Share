# app.py # Single-file app: runs a server, serves the page, accepts multiple files, # zips them into a single bundle, and returns a QR link that downloads all. # Usage: # pip install Flask # python app.py # Then open http://:5000 (QR will use window.location.origin) import os, uuid, datetime from zipfile import ZipFile, ZIP_DEFLATED from flask import Flask, request, jsonify, send_from_directory, render_template_string APP_PORT = 5000 BASE_DIR = os.path.abspath(os.path.dirname(__file__)) MEDIA_ROOT = os.path.join(BASE_DIR, "media") UPLOADS_DIR = os.path.join(MEDIA_ROOT, "uploads") BUNDLES_DIR = os.path.join(MEDIA_ROOT, "bundles") os.makedirs(UPLOADS_DIR, exist_ok=True) os.makedirs(BUNDLES_DIR, exist_ok=True) app = Flask(__name__, static_url_path="/media", static_folder=MEDIA_ROOT) PAGE = r""" üì° Smart File Share

üì° Smart File Share

Seamless transfers between your Laptop and Phone

üì§ Send Files to Phone

üì• Receive Files from Phone

Scan on Phone to open this page:

Created by Alvin Shedrack ¬© 2025
""" def _zip_files(saved_paths): """Create a zip in BUNDLES_DIR and return (bundle_rel_path, download_name).""" bundle_id = uuid.uuid4().hex ts = datetime.datetime.now().strftime("%Y-%m-%d_%H-%M") download_name = f"share-{ts}.zip" bundle_rel = os.path.join("bundles", f"{bundle_id}.zip") bundle_abs = os.path.join(MEDIA_ROOT, bundle_rel) with ZipFile(bundle_abs, mode="w", compression=ZIP_DEFLATED) as zf: for abs_path, original_name in saved_paths: zf.write(abs_path, arcname=original_name) return bundle_rel.replace("\\", "/"), download_name @app.route("/", methods=["GET"]) def index(): return render_template_string(PAGE) @app.route("/upload", methods=["POST"]) def upload(): """ Accept multiple input fields named 'file'. Save to media/uploads/, zip them into media/bundles/.zip, and return JSON with a single bundle URL that downloads all files. """ files = request.files.getlist("file") if not files: return jsonify({"error": "no files"}), 400 saved = [] for f in files: original = f.filename or "file" # handle name collisions by prefixing a short uuid base, ext = os.path.splitext(original) safe_name = f"{base}_{uuid.uuid4().hex[:6]}{ext}" if os.path.exists(os.path.join(UPLOADS_DIR, original)) else original abs_path = os.path.join(UPLOADS_DIR, safe_name) f.save(abs_path) saved.append((abs_path, original)) # store original name inside zip bundle_rel, download_name = _zip_files(saved) bundle_url = f"/media/{bundle_rel}" return jsonify({ "files": [{"filename": os.path.basename(p)} for p, _ in saved], "bundle": {"url": bundle_url, "filename": download_name} }) @app.route("/media/") def media(filename): # Serve uploaded files and bundles return send_from_directory(MEDIA_ROOT, filename, as_attachment=False) if __name__ == "__main__": # 0.0.0.0 to be reachable on LAN; open http://:5000 app.run(host="0.0.0.0", port=APP_PORT, debug=True)
